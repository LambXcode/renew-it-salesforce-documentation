    @RestResource(urlMapping='/Account/*')
global with sharing class AccountRESTWebservice {

    global static List<string> restServiceErrorMessages{get;set;}

    global class ClientReqModel{   
        public string Id {get;set;}
        public string Name {get;set;}
        public string LastName {get;set;}
        public string FirstName {get;set;}
        public string MiddleName {get;set;}
        public string Salutation {get;set;}
        public string Suffix {get;set;}
        public string TYPE {get;set;}
        public string RECORDTYPEID {get; set;}        
        public string PERSONMOBILEPHONE {get;set;}
        public string PERSONEMAIL {get;set;}
        public string PREFERRED_COMMUNICATION_METHOD {get;set;}
        public string OPT_IN_FOR_AFTER_SALES_MARKETING {get;set;}
        public string OTHER_INFORMATION {get;set;}   
        public string IsPersonAccount {get; set;} 
        public string Mobile {get; set;}
        public string Email {get; set;} 
        public string PrefComm {get; set;} 
        public string Optin {get; set;} 
        public string Other {get; set;} 
        public boolean Insurer {get; set;} 
        public boolean Broker {get; set;} 
        public boolean Admin {get; set;}
        public boolean Estimator {get; set;}
        public boolean ServiceAdvisor {get; set;}
        public integer ClaimID {get; set;}
        public integer BranchID {get; set;}
        public string SalesForceID {get; set;}
        public string DateAdded {get; set;}
    }

    global class ClientReqResponseModel{   
        public string SFClientUniqueId { get; set; }
        public string ClientName {get;set;}
        public string ClientContactNo {get;set;}
        public string MethodName { get; set; }
        public string WasSuccessful { get; set; }
        public string ErrorMessage { get; set; }
    }

    @HttpPost
    global static ClientReqResponseModel CreateClient(ClientReqModel clientModel) {
        ClientReqResponseModel response = new ClientReqResponseModel();
        List<String> restServiceErrorMessages = new List<String>();

        try {
            System.debug('Input clientModel: ' + JSON.serializePretty(clientModel));

            if (clientModel == null) {
                throw new AuraHandledException('Client model is null');
            }

            Boolean isInsurer = clientModel.Insurer != null ? clientModel.Insurer : false;
            Boolean isBroker = clientModel.Broker != null ? clientModel.Broker : false;
            Boolean isAdmin = clientModel.Admin != null ? clientModel.Admin : false;
            Boolean isEstimator = clientModel.Estimator != null ? clientModel.Estimator : false;
            Boolean IsServiceAdvisor = clientModel.ServiceAdvisor != null ? clientModel.ServiceAdvisor : false;

            if (isInsurer || isBroker) {

                List<Account> existingAccounts = [
                    SELECT Id, Name, Phone
                    FROM Account
                    WHERE Name = :(clientModel.FirstName != null ? clientModel.FirstName : '') + ' ' + (clientModel.LastName != null ? clientModel.LastName : '')
                    LIMIT 1
                ];

                Account account;
                if (!existingAccounts.isEmpty()) {

                    account = existingAccounts[0];
                    response.SFClientUniqueId = account.Id;
                    response.ClientName = account.Name;
                    response.ClientContactNo = account.Phone;
                    response.WasSuccessful = 'true';
                    response.MethodName = 'CreateClient - Existing Found';

                } else {

                    account = new Account();
                    account.Name = (clientModel.FirstName != null ? clientModel.FirstName : '') + ' ' + (clientModel.LastName != null ? clientModel.LastName : '');

                    if (String.isBlank(account.Name)) {
                        throw new AuraHandledException('Account name cannot be blank');
                    }

                    if (isInsurer) {
                        account.RecordTypeId = Schema.SObjectType.Account.getRecordTypeInfosByDeveloperName().get('Insurance_Company').getRecordTypeId();
                        account.Type = String.isNotBlank(clientModel.Type) ? clientModel.Type : 'Insurance Company';
                    } else if (isBroker) {
                        account.RecordTypeId = Schema.SObjectType.Account.getRecordTypeInfosByDeveloperName().get('Broker').getRecordTypeId();
                        account.Type = String.isNotBlank(clientModel.Type) ? clientModel.Type : 'Broker'; 
                    }

                    account.Phone = String.isNotBlank(clientModel.Mobile) ? clientModel.Mobile : (String.isNotBlank(clientModel.PERSONMOBILEPHONE) ? clientModel.PERSONMOBILEPHONE : '');

                    System.debug('Account to be inserted: ' + JSON.serializePretty(account));
                    insert account;

                    response.SFClientUniqueId = account.Id;
                    response.ClientName = account.Name;
                    response.ClientContactNo = account.Phone;
                    response.WasSuccessful = 'true';
                    response.MethodName = 'CreateClient';
                }
            } else {

                List<Contact> existingContacts = [
                    SELECT Id, FirstName, LastName, MobilePhone
                    FROM Contact
                    WHERE FirstName = :clientModel.FirstName
                    AND LastName = :clientModel.LastName
                    LIMIT 1
                ];

                Contact client;
                if (!existingContacts.isEmpty()) {

                    client = existingContacts[0];
                    response.SFClientUniqueId = client.Id;
                    response.ClientName = client.FirstName + ' ' + client.LastName;
                    response.ClientContactNo = client.MobilePhone;
                    response.WasSuccessful = 'true';
                    response.MethodName = 'CreateClient - Existing Found';
                } else {

                    client = new Contact();
                    client.FirstName = clientModel.FirstName;
                    client.LastName = clientModel.LastName;
                    client.MiddleName = clientModel.MiddleName;
                    client.Salutation = clientModel.Salutation;
                    client.Email = String.isNotBlank(clientModel.Email) ? clientModel.Email : (String.isNotBlank(clientModel.PERSONEMAIL) ? clientModel.PERSONEMAIL : ' ');
                    client.MobilePhone = String.isNotBlank(clientModel.Mobile) ? clientModel.Mobile : clientModel.PERSONMOBILEPHONE;
                    client.Phone = String.isNotBlank(clientModel.Mobile) ? clientModel.Mobile : (String.isNotBlank(clientModel.PERSONMOBILEPHONE) ? clientModel.PERSONMOBILEPHONE : '');
                    client.OPT_In_For_After_Sales_Marketing__c = String.isNotBlank(clientModel.Optin) ? clientModel.Optin : clientModel.OPT_IN_FOR_AFTER_SALES_MARKETING;
                    client.Other_Information__c = String.isNotBlank(clientModel.Other) ? clientModel.Other : clientModel.OTHER_INFORMATION;
                    client.Preferred_Communication_Method__c = String.isNotBlank(clientModel.PrefComm) ? clientModel.PrefComm : clientModel.PREFERRED_COMMUNICATION_METHOD;

                    if (isAdmin) {
                        client.RecordTypeId = Schema.SObjectType.Contact.getRecordTypeInfosByDeveloperName().get('Claims_Administrator').getRecordTypeId();
                        client.Employee_Type__c = 'Claims Handler';
                    } else if (isEstimator) {
                        client.RecordTypeId = Schema.SObjectType.Contact.getRecordTypeInfosByDeveloperName().get('Branch_Employee').getRecordTypeId();
                        client.Employee_Type__c = 'Estimator';
                    } else if (IsServiceAdvisor) {
                        client.RecordTypeId = Schema.SObjectType.Contact.getRecordTypeInfosByDeveloperName().get('Branch_Employee').getRecordTypeId();
                        client.Employee_Type__c = 'CSA';
                    } else {
                        client.RecordTypeId = Schema.SObjectType.Contact.getRecordTypeInfosByDeveloperName().get('Client').getRecordTypeId();
                        client.Type__c = 'Customer';
                    }

                    System.debug('Contact to be inserted: ' + JSON.serializePretty(client));
                    insert client;

                    response.SFClientUniqueId = client.Id;
                    response.ClientName = client.FirstName + ' ' + client.LastName;
                    response.ClientContactNo = client.MobilePhone;
                    response.WasSuccessful = 'true';
                    response.MethodName = 'CreateClient';
                }
            }
        } catch (Exception ex) {
            System.debug('ERROR on CreateClient: ' + ex.getMessage() + '\n' + ex.getStackTraceString());
            response.ErrorMessage = ex.getMessage();
            response.MethodName = 'CreateClient';
            response.SFClientUniqueId = null;
            response.WasSuccessful = 'false';
        }

        System.debug('Response: ' + JSON.serializePretty(response));
        return response;
    }

    @HttpPut
    global static ClientReqResponseModel UpdateClient(ClientReqModel clientModel) {    
        ClientReqResponseModel respModel = new ClientReqResponseModel();        

        try {
            System.debug('Input clientModel for update: ' + JSON.serializePretty(clientModel));

            if (clientModel == null || String.isBlank(clientModel.Id)) {
                throw new IllegalArgumentException('Invalid client model or missing ID');
            }

            String idPrefix = clientModel.Id.substring(0, 3);

            if (idPrefix == '001') {  

                Account account = [SELECT Id, Name, RecordType.DeveloperName, Phone, Type, IsPersonAccount,
                                   FirstName, LastName, PersonEmail, PersonMobilePhone
                                   FROM Account 
                                   WHERE Id = :clientModel.Id];

                System.debug('Existing Account before update: ' + JSON.serializePretty(account));

                if (account.IsPersonAccount) {
                    account.FirstName = String.isNotBlank(clientModel.FirstName) ? clientModel.FirstName : account.FirstName;
                    account.LastName = String.isNotBlank(clientModel.LastName) ? clientModel.LastName : account.LastName;
                    account.PersonEmail = String.isNotBlank(clientModel.PERSONEMAIL) ? clientModel.PERSONEMAIL : (String.isNotBlank(clientModel.Email) ? clientModel.Email : ' ');
                    account.PersonMobilePhone = String.isNotBlank(clientModel.PERSONMOBILEPHONE) ? clientModel.PERSONMOBILEPHONE : (String.isNotBlank(clientModel.Mobile) ? clientModel.Mobile : ' ');
                } else {

                    String name = '';
                    if (String.isNotBlank(clientModel.FirstName)) {
                        name += clientModel.FirstName;
                    }
                    if (String.isNotBlank(clientModel.LastName)) {
                        if (name.length() > 0) {
                            name += ' ';
                        }
                        name += clientModel.LastName;
                    }
                    account.Name = name;

                    account.Phone = String.isNotBlank(clientModel.Mobile) ? clientModel.Mobile : (String.isNotBlank(clientModel.PERSONMOBILEPHONE) ? clientModel.PERSONMOBILEPHONE : '');
                }

                System.debug('Account to be updated: ' + JSON.serializePretty(account));
                update account;

                respModel.SFClientUniqueId = account.Id;
                respModel.ClientName = account.IsPersonAccount ? (account.FirstName + ' ' + account.LastName) : account.Name;
                respModel.ClientContactNo = account.IsPersonAccount ? account.PersonMobilePhone : account.Phone;
                respModel.WasSuccessful = 'true';

            } else if (idPrefix == '003') { 

                Contact contact = [SELECT Id, FirstName, LastName, MiddleName, Salutation, MobilePhone, Phone, Email,
                                   OPT_In_For_After_Sales_Marketing__c, Other_Information__c,
                                   Preferred_Communication_Method__c, Type__c, RecordType.DeveloperName, Employee_Type__c
                                   FROM Contact WHERE Id = :clientModel.Id];

                System.debug('Existing Contact before update: ' + JSON.serializePretty(contact));

                contact.FirstName = String.isNotBlank(clientModel.FirstName) ? clientModel.FirstName : contact.FirstName;
                contact.LastName = String.isNotBlank(clientModel.LastName) ? clientModel.LastName : contact.LastName;
                contact.MiddleName = String.isNotBlank(clientModel.MiddleName) ? clientModel.MiddleName : contact.MiddleName;
                contact.Salutation = String.isNotBlank(clientModel.Salutation) ? clientModel.Salutation : contact.Salutation;
                contact.MobilePhone = String.isNotBlank(clientModel.PERSONMOBILEPHONE) ? clientModel.PERSONMOBILEPHONE : (String.isNotBlank(clientModel.Mobile) ? clientModel.Mobile : contact.MobilePhone);
                contact.Phone = String.isNotBlank(clientModel.PERSONMOBILEPHONE) ? clientModel.PERSONMOBILEPHONE : (String.isNotBlank(clientModel.Mobile) ? clientModel.Mobile : contact.Phone);
                contact.Email = String.isNotBlank(clientModel.PERSONEMAIL) ? clientModel.PERSONEMAIL : (String.isNotBlank(clientModel.Email) ? clientModel.Email : ' ');
                contact.OPT_In_For_After_Sales_Marketing__c = String.isNotBlank(clientModel.OPT_IN_FOR_AFTER_SALES_MARKETING) ? clientModel.OPT_IN_FOR_AFTER_SALES_MARKETING : clientModel.Optin;
                contact.Other_Information__c = String.isNotBlank(clientModel.OTHER_INFORMATION) ? clientModel.OTHER_INFORMATION : clientModel.Other;
                contact.Preferred_Communication_Method__c = String.isNotBlank(clientModel.PREFERRED_COMMUNICATION_METHOD) ? clientModel.PREFERRED_COMMUNICATION_METHOD : clientModel.PrefComm;

                System.debug('Contact to be updated: ' + JSON.serializePretty(contact));
                update contact;

                respModel.SFClientUniqueId = contact.Id;
                respModel.ClientName = contact.FirstName + ' ' + contact.LastName;
                respModel.ClientContactNo = contact.MobilePhone;
                respModel.WasSuccessful = 'true';

            } else {
                throw new IllegalArgumentException('Invalid ID prefix. Expected Account or Contact ID.');
            }

            respModel.ErrorMessage = '';
            respModel.MethodName = 'UpdateClient';

        } catch (Exception ex) {
            System.debug('ERROR on UpdateClient: ' + ex.getMessage() + '\n' + ex.getStackTraceString());
            respModel.ErrorMessage = ex.getMessage();
            respModel.MethodName = 'UpdateClient';
            respModel.SFClientUniqueId = clientModel.Id;
            respModel.WasSuccessful = 'false';
        } 

        System.debug('Update Response: ' + JSON.serializePretty(respModel));
        return respModel;
    }
